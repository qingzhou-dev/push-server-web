<script setup lang="ts">
import { computed, onMounted, ref, watch } from 'vue'
import { ElMessage, ElMessageBox } from 'element-plus'
import { RefreshRight } from '@element-plus/icons-vue'
import {
  createAppApiKey,
  deleteAppApiKey,
  fetchAppApiKey,
  fetchApps,
  updateAppApiKey,
} from '@/api/apps'
import type {
  PortalAppApiKeyResponse,
  PortalAppResponse,
} from '@/api/types'

const apps = ref<PortalAppResponse[]>([])
const selectedAppId = ref<number | null>(null)
const apiKeyInfo = ref<PortalAppApiKeyResponse | null>(null)
const isLoadingApps = ref(false)
const isKeyLoading = ref(false)
const isMutating = ref(false)
const rateLimit = ref<number | null>(null)

const selectedApp = computed(() =>
  apps.value.find((app) => app.id === selectedAppId.value),
)
const hasKey = computed(() => apiKeyInfo.value?.hasKey ?? false)
const apiKeyValue = computed(() => apiKeyInfo.value?.apiKey || '')

const formatDate = (value?: number | null) => {
  if (!value) return '--'
  return new Date(value).toLocaleString()
}

const loadApps = async () => {
  isLoadingApps.value = true
  try {
    apps.value = await fetchApps()
    if (!selectedAppId.value && apps.value.length) {
      selectedAppId.value = apps.value[0].id
    }
  } catch (error) {
    ElMessage.error('è·å–åº”ç”¨åˆ—è¡¨å¤±è´¥')
  } finally {
    isLoadingApps.value = false
  }
}

const loadApiKey = async (appId: number) => {
  isKeyLoading.value = true
  try {
    apiKeyInfo.value = await fetchAppApiKey(appId)
    rateLimit.value = apiKeyInfo.value.rateLimitPerMinute ?? null
  } catch (error) {
    apiKeyInfo.value = null
    rateLimit.value = null
  } finally {
    isKeyLoading.value = false
  }
}

const handleGenerate = async () => {
  if (!selectedAppId.value) {
    ElMessage.warning('è¯·å…ˆé€‰æ‹©åº”ç”¨')
    return
  }
  isMutating.value = true
  try {
    await createAppApiKey(selectedAppId.value)
    ElMessage.success('å·²ç”Ÿæˆæ–°çš?API Key')
    loadApiKey(selectedAppId.value)
  } catch (error) {
    ElMessage.error('ç”Ÿæˆå¯†é’¥å¤±è´¥')
  } finally {
    isMutating.value = false
  }
}

const handleDelete = async () => {
  if (!selectedAppId.value) return
  try {
    await ElMessageBox.confirm(
      'åˆ é™¤åè¯¥åº”ç”¨çš„å¼€æ”¾æ¥å£å°†æ— æ³•è°ƒç”¨ï¼Œç¡®è®¤åˆ é™¤å½“å‰å¯†é’¥å—ï¼?,
      'åˆ é™¤ç¡®è®¤',
      {
        type: 'warning',
        confirmButtonText: 'åˆ é™¤',
        cancelButtonText: 'å–æ¶ˆ',
      },
    )
  } catch (error) {
    return
  }

  isMutating.value = true
  try {
    await deleteAppApiKey(selectedAppId.value)
    ElMessage.success('å·²åˆ é™?API Key')
    apiKeyInfo.value = null
    rateLimit.value = null
  } catch (error) {
    ElMessage.error('åˆ é™¤å¤±è´¥')
  } finally {
    isMutating.value = false
  }
}

const handleUpdateRateLimit = async () => {
  if (!selectedAppId.value) return
  isMutating.value = true
  try {
    await updateAppApiKey(selectedAppId.value, {
      rateLimitPerMinute: rateLimit.value ?? null,
    })
    ElMessage.success('å·²æ›´æ–°é™æµé…ç½?)
    loadApiKey(selectedAppId.value)
  } catch (error) {
    ElMessage.error('æ›´æ–°é™æµå¤±è´¥')
  } finally {
    isMutating.value = false
  }
}

const handleCopyKey = async () => {
  if (!apiKeyValue.value) return
  try {
    await navigator.clipboard.writeText(apiKeyValue.value)
    ElMessage.success('å·²å¤åˆ?API Key')
  } catch (error) {
    ElMessage.error('å¤åˆ¶å¤±è´¥ï¼Œè¯·æ‰‹åŠ¨é€‰æ‹©å¤åˆ¶')
  }
}

onMounted(() => {
  loadApps()
})

watch(selectedAppId, (appId) => {
  if (appId) {
    loadApiKey(appId)
  } else {
    apiKeyInfo.value = null
  }
})
</script>

<template>
  <section class="page">
    <div class="page-heading">
      <div>
        <h1>API å¯†é’¥</h1>
        <p>ä¸ºæ¯ä¸ªåº”ç”¨ç®¡ç†å¼€æ”¾æ¥å£çš„ API Keyï¼Œæ”¯æŒç”Ÿæˆ?é‡ç½®ã€é™æµå’Œåˆ é™¤ã€?/p>
      </div>
      <div class="page-actions">
        <el-button :icon="RefreshRight" @click="loadApps">åˆ·æ–°åˆ—è¡¨</el-button>

                <p>é€‰æ‹©éœ€è¦æŸ¥çœ‹æˆ–æ“ä½œå¯†é’¥çš„åº”ç”?/p>
              </div>

            <el-form-item label="åº”ç”¨">

                  :value="app.id"
                />
              </el-select>
            </el-form-item>
            <el-form-item label="AgentId">
              <el-input
                :model-value="selectedApp?.agentId || ''"

              />
            </el-form-item>
            <el-form-item label="åç§°">
              <el-input
                :model-value="selectedApp?.name || '--'"
                disabled

              />
            </el-form-item>
          </el-form>
        </el-card>

      <el-col :xs="24" :lg="12">
        <el-card class="panel-card">
          <template #header>
            <div class="panel-header">
              <div>
                <h3>å¯†é’¥è¯¦æƒ…</h3>
                <p>ç”Ÿæˆã€å¤åˆ¶æˆ–åˆ é™¤å½“å‰åº”ç”¨çš?API Key</p>
              </div>
              <el-button
                :icon="RefreshRight"
                text
                :loading="isKeyLoading"
                @click="selectedAppId && loadApiKey(selectedAppId)"
              >
                åˆ·æ–°
              </el-button>
            </div>
          </template>

          <el-skeleton

          >
            <template #default>
              <el-alert
                v-if="selectedAppId && !hasKey"
                title="å½“å‰åº”ç”¨å°šæœªç”Ÿæˆ API Key"
                type="info"
                show-icon
                class="mb-16"
                description="ç”Ÿæˆåå³å¯é€šè¿‡ OpenAPI è°ƒç”¨æ¶ˆæ¯å‘é€æ¥å£ã€?
              />

              <el-alert
                v-else-if="!selectedAppId"
                title="è¯·é€‰æ‹©åº”ç”¨ä»¥æŸ¥çœ‹å¯†é’?
                type="warning"
                show-icon
                class="mb-16"
              />

              <el-form label-width="110px">
                <el-form-item label="å½“å‰ Key">
                  <el-input

                <el-form-item label="æ¯åˆ†é’Ÿé™æµ?>

                </el-form-item>

                <el-form-item>
                  <el-button
                    type="primary"
                    :loading="isMutating"
                    :disabled="!selectedAppId"
                    @click="handleGenerate"
                  >
                    {{ hasKey ? 'é‡ç½®/ç”Ÿæˆæ–?Key' : 'ç”Ÿæˆ API Key' }}
                  </el-button>
                  <el-button
                    :loading="isMutating"
                    :disabled="!selectedAppId"
                    @click="handleUpdateRateLimit"
                  >
                    æ›´æ–°é™æµ
                  </el-button>
                  <el-button
                    type="danger"
                    :loading="isMutating"
                    :disabled="!selectedAppId || !hasKey"
                    @click="handleDelete"
                  >
                    åˆ é™¤å¯†é’¥
                  </el-button>
                </el-form-item>
              </el-form>
            </template>
          </el-skeleton>
        </el-card>
      </el-col>
    </el-row>
  </section>
